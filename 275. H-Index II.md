### 代码
```c++
class Solution {
public:
    int hIndex(vector<int>& citations) {
        int min_fre=INT_MAX,chapter=0;
        for(int i=citations.size()-1;i>=0;i--){
            chapter++;
            min_fre=citations[i];
            if(min_fre<chapter) return chapter-1;
        }
        return chapter;
    }
};
```
