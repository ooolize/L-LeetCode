### 题目
[300. Longest Increasing Subsequence](https://leetcode-cn.com/problems/longest-increasing-subsequence/)
### 思路
52ms 67,36%

动态规划，a[i]=a[j]+1,
### 代码
```c++
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n=nums.size(),max_ans=1;
        if(n==0)return 0;
        vector<int> a(n,1);
        a[0]=1;
        
        for(int i=1;i<n;i++){
            int max_val=0;
            for(int j=0;j<i;j++){
                if(nums[j]<nums[i])  
                    max_val=max(max_val,a[j]);
            }
            a[i]=max_val+1;
            max_ans=max(max_ans,a[i]);
        }
        
        return max_ans;
    }
};
```
