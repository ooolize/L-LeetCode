### 代码
位运算
```c++
class Solution {
public:
    int integerReplacement(int x) {
        long n=x;
        int count=0;
        while(n!=1){
            if((n&1)==0){
                n>>=1;
                count++;
            }
            else {
                if((n&2)==0){
                    n--;
                    count++;
                }
                else{
                    if(n==3){count+=2;break;}
                    n++;
                    count++;
                }
            }
        }
        return count;
    }
};
```

class Solution {
public:
    int integerReplacement(int n) {
         vector<int> v(n+1,0);
        v[1]=0;
        
        for(int i=2;i<=n;i++){
            if(i%2==0) v[i]=v[i/2]+1;
            else
                v[i]=min(v[i-1],v[(i+1)/2]+1)+1;
        }
        return v[n];
    }
};
