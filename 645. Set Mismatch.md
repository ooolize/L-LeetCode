### 题目
[645. Set Mismatch](https://leetcode-cn.com/problems/set-mismatch/submissions/)
### 思路
44ms 61.91%

注意返回{a,b},不是{b,a}
### 代码

```c++
class Solution {
public:
    vector<int> findErrorNums(vector<int>& nums) {
        int n=nums.size(),a,b;
        vector<int> vs(n+1,-1);
        for(auto& p:nums){
            vs[p]++;
        }
        for(int i=1;i<=n;i++){
            if(vs[i]==-1) a=i;
            else if(vs[i]==1) b=i;
        }
        
        return{b,a};
    }
};
```
