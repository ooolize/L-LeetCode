class Solution {
public:
    int maxRotateFunction(vector<int>& A) {
        int ans=0,total=0,n=A.size();
        if(n==0) return 0;
        vector<int> a(n,0);
        total=accumulate(A.begin(),A.end(),0);
        
        for(int i=0;i<n;i++)
            a[n-1]+=i*A[i];
        
        ans=a[n-1];
        for(int i=n-2;i>=0;i--){
            if(ans==2147483647) {break;}
            a[i]=a[i+1]+(total-A[i+1])-(n-1)*A[i+1];
            ans=max(a[i],ans);
        }
        return ans;
    }
};
