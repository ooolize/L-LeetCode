### 代码
暴力模拟
```c++
class Solution {
public:
    int minMoves2(vector<int>& nums) {
       long count=INT_MAX;
       for(auto p:nums){
           long tmp=0;
           for(auto q:nums){
               tmp+=abs((long)(p-q));
           }
           count=min(count,tmp);
       }
        return count;
    }
};
```
