```c++
class Solution {
public:
    int lastRemaining(int n) {
        vector<int> v(n,0),tmp;
        
        iota(v.begin(),v.end(),1);
        while(v.size()!=1){
            for(int i=1;i<v.size();i+=2){
                tmp.push_back(v[i]);
            }
            reverse(tmp.begin(),tmp.end());
            v=tmp;
            tmp.clear();
        }
        return v.back();
    }
};
```

class Solution {
public:
    int lastRemaining(int n) {
        if(n==1) return 1;
        return 2*(n/2-lastRemaining(n/2)+1);
    }
};
