### 题目
[589. N-ary Tree Preorder Traversal](https://leetcode-cn.com/problems/n-ary-tree-preorder-traversal/)

### 代码
```c++
class Solution {
public:
    void VLR(Node* root,vector<int>&v){
        if(!root)return;
        v.push_back(root->val);
        for(auto p:root->children){
            VLR(p,v);
        }
        return;
    }
    
    vector<int> preorder(Node* root) {
        vector<int> v;
        if(root)VLR(root,v);
        return v;
    }
};
```
