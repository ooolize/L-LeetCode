### 题目
[897. Increasing Order Search Tree](https://leetcode-cn.com/problems/surface-area-of-3d-shapes/submissions/)

### 代码
```c++
class Solution {
public:
    TreeNode* res;
    TreeNode* increasingBST(TreeNode* root) {
        res = new TreeNode(0);
        TreeNode* t = res;
        helper(root);
        return t->right;
    }
    void helper(TreeNode* root) {
        if(!root) {
            return;
        }
        helper(root->left);
        TreeNode* n = new TreeNode(root->val);
        res->right = n;
        res = n;
        helper(root->right);
        return;
    }
};
```
