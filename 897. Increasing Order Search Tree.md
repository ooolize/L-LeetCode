### 代码
```c++
class Solution {
public:
    TreeNode* res;
    TreeNode* increasingBST(TreeNode* root) {
        res = new TreeNode(0);
        TreeNode* t = res;
        helper(root);
        return t->right;
    }
    void helper(TreeNode* root) {
        if(!root) {
            return;
        }
        helper(root->left);
        TreeNode* n = new TreeNode(root->val);
        res->right = n;
        res = n;
        helper(root->right);
        return;
    }
};
```
这样为什么不对
```c++
class Solution {
public:
    void dfs(TreeNode*root,TreeNode* tmp){
        if(!root)return;
        
        dfs(root->left,tmp);
        
        TreeNode* n=new TreeNode(root->val);
        tmp->right=n;
        tmp=tmp->right;
        
        dfs(root->right,tmp);
        return;
    }
    
    TreeNode* increasingBST(TreeNode* root) {
        TreeNode* dummy=new TreeNode(-1),*tmp=dummy;
        dfs(root,tmp);
        return dummy->right;
    }
};
```
