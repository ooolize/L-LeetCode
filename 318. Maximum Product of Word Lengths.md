### 题目
[318. Maximum Product of Word Lengths](https://leetcode-cn.com/problems/maximum-product-of-word-lengths/)
### 思路
本来是想暴力解法 结果超时
### 代码
```c++
class Solution {
public:
    bool no_equal(string &s1, string &s2){
	 for (auto &p : s1){
		 for (auto &q : s2){
			 if (p == q)
				 return false;
		 }
	 }
	 return true;
 }

 int maxProduct(vector<string>& words) {
	 int n = words.size(),ans=0;
     if(n==0)return 0;
	 vector<int>v;

	 for (auto&p : words)
		 v.push_back(p.size());

	 for (int i = 1; i<n; i++){
         
		 for (int j = i - 1; j >= 0; j--){
			 if (no_equal(words[j], words[i]))
				 ans = max(ans, v[i] * v[j]);
		 }
	 }
	 return ans;
 } 
};
```
