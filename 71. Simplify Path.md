### 代码
```c++
class Solution {
public:
     string simplifyPath(string path) {
	 stack<string> s;
	 s.push("/");

	 string result(""), tmp;
	 for (int i = 0; i<path.size();){
		 if (path[i] == '/'){
			 if (s.top() == "/") { i++; continue; }
			 else { s.push("/"); i++; continue; }
		 }
		 tmp += path[i++];
		 if (i == path.size() || path[i] == '/'){
			 if (tmp == ".")  {}
			 else if (tmp == ".."){
				 if (s.size() == 1 && s.top() == "/") {}
				 else {
					 s.pop(); s.pop(); 
				 }
			 }
			 else s.push(tmp);
			 tmp = "";
		 }
	 }
	 if (s.top() == "/") s.pop();

	 while (!s.empty()){
		 result = s.top() + result;
		 s.pop();
	 }

	 return result == "" ? "/" : result;
 } 

};
```
