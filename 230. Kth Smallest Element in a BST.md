### 代码
 AC代码
```c++
class Solution {
public:
    void LVR(TreeNode* root,int&k,int &ans){
        if(root==NULL) return;
        LVR(root->left,k,ans);
        if(ans==NULL&&--k==0) ans=root->val;
        if(ans==NULL) LVR(root->right,k,ans);
        return;
    }
    
    int kthSmallest(TreeNode* root, int k) {
        int ans=0;
        LVR(root,k,ans);
        return ans;
    }
};
```
